Inferring coupling strengths from configuration of variables¶

Method: Free Energy Minimization (FEM)¶

In [1]:

import numpy as np
import sys
import matplotlib.pyplot as plt
import myfunction as mf
#import meanfield as meanfield
import fem as fem
import myplot as mp

np.random.seed(1)

# parameter:
n = 100    # number of variables

def check_performance(g,l):
    # simulate data:
    w0,s0 = mf.simulate_data(n,g,l)

    # recover couplings from observed data s0 by using Free Energy Minimization (FEM)
    wFEM = fem.fit_interaction(s0)
    mp.plot_w(w0,wFEM)
